function TOGGLE_COMBO()
    gg.setRanges(gg.REGION_C_ALLOC | gg.REGION_ANONYMOUS)

    if SL == off then
        gg.searchNumber("h 00 00 80 40 33 33 93 40 3D 0A F7 3F", gg.TYPE_BYTE)
        gg.getResults(100)
        gg.editAll("h 00 00 80 40 00 00 80 40 CB D2 4D 3E", gg.TYPE_BYTE)
        gg.clearResults()
        gg.toast("✅ SLOW MOVEMENT ON")
        SL = on
    elseif SL == on then

        gg.getResults(100)
        gg.editAll("h 00 00 80 40 33 33 93 40 3D 0A F7 3F", gg.TYPE_BYTE)
        gg.clearResults()
        gg.toast("❌ SLOW MOVEMENT OFF")
        SL = off
    end
end
